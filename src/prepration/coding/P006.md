# Question 1: Two Sum

### Problem

Given an array and a target, return indices whose sum equals target.

```javascript
Input: nums = [2, 7, 11, 15];
target = 9;

Output: [0, 1];
```

### Solution

```javascript
function twoSum(nums, target) {
  const map = {};

  for (let i = 0; i < nums.length; i++) {
    const diff = target - nums[i];

    if (map[diff] !== undefined) {
      return [map[diff], i];
    }

    map[nums[i]] = i;
  }

  return [];
}
```

Time: O(n)

# Question 2: Valid Anagram

### Problem

```javascript
Input: s = "listen";
t = "silent";

Output: true;
```

### Solution

```javascript
function isAnagram(s, t) {
  if (s.length !== t.length) return false;

  const map = {};

  for (const ch of s) {
    map[ch] = (map[ch] || 0) + 1;
  }

  for (const ch of t) {
    if (!map[ch]) return false;
    map[ch]--;
  }

  return true;
}
```

Time: O(n)

### Problem

```javascript
Input: "swiss";

Output: "w";
```

### Solution

```javascript
function firstUnique(str) {
  const freq = {};

  for (const ch of str) {
    freq[ch] = (freq[ch] || 0) + 1;
  }

  for (const ch of str) {
    if (freq[ch] === 1) {
      return ch;
    }
  }

  return null;
}
```

Time: O(n)

# Question 4: Move Zeroes

### Problem

```javascript
Input: [0, 1, 0, 3, 12];

Output: [1, 3, 12, 0, 0];
```

### Solution

```javascript
function moveZeroes(nums) {
  let index = 0;

  for (let i = 0; i < nums.length; i++) {
    if (nums[i] !== 0) {
      nums[index++] = nums[i];
    }
  }

  while (index < nums.length) {
    nums[index++] = 0;
  }

  return nums;
}
```

Time: O(n)

# Question 5: Longest Substring Without Repeating Characters

### Problem

```javascript
Input: "abcabcbb";

Output: 3;
```

### Solution

```javascript
function lengthOfLongestSubstring(str) {
  let left = 0;
  let max = 0;

  const set = new Set();

  for (let right = 0; right < str.length; right++) {
    while (set.has(str[right])) {
      set.delete(str[left]);
      left++;
    }

    set.add(str[right]);

    max = Math.max(max, right - left + 1);
  }

  return max;
}
```

Time: O(n)

# Question 6: Valid Parentheses

### Problem

```javascript
Input: "{[()]}";

Output: true;
```

### Solution

```javascript
function isValid(str) {
  const stack = [];

  const map = {
    ")": "(",
    "]": "[",
    "}": "{",
  };

  for (const ch of str) {
    if (ch === "(" || ch === "[" || ch === "{") {
      stack.push(ch);
    } else {
      if (stack.pop() !== map[ch]) {
        return false;
      }
    }
  }

  return stack.length === 0;
}
```

Time: O(n)

# Question 7: Maximum Subarray Sum (Kadane)

### Problem

```javascript
Input: [-2, 1, -3, 4, -1, 2, 1, -5, 4];

Output: 6;
```

### Solution

```javascript
function maxSubArray(nums) {
  let current = nums[0];
  let max = nums[0];

  for (let i = 1; i < nums.length; i++) {
    current = Math.max(nums[i], current + nums[i]);

    max = Math.max(max, current);
  }

  return max;
}
```

Time: O(n)

# Question 8: Product Of Array Except Self

### Problem

```javascript
Input: [1, 2, 3, 4];

Output: [24, 12, 8, 6];
```

### Solution

```javascript
function productExceptSelf(nums) {
  const result = new Array(nums.length).fill(1);

  let prefix = 1;

  for (let i = 0; i < nums.length; i++) {
    result[i] = prefix;
    prefix *= nums[i];
  }

  let suffix = 1;

  for (let i = nums.length - 1; i >= 0; i--) {
    result[i] *= suffix;
    suffix *= nums[i];
  }

  return result;
}
```

Time: O(n)

# Question 9: Group Anagrams

### Problem

```javascript
Input: ["eat", "tea", "tan", "ate", "nat", "bat"];
```

Output

```javascript
[["eat", "tea", "ate"], ["tan", "nat"], ["bat"]];
```

### Solution

```javascript
function groupAnagrams(strs) {
  const map = {};

  for (const word of strs) {
    const key = word.split("").sort().join("");

    if (!map[key]) {
      map[key] = [];
    }

    map[key].push(word);
  }

  return Object.values(map);
}
```

Time: O(n \* k log k)

# Question 10: Flatten Nested Array

### Problem

```javascript
Input: [1, [2, [3, [4]], 5]];

Output: [1, 2, 3, 4, 5];
```

### Solution

### Recursive

```javascript
function flatten(arr) {
  let result = [];

  for (const item of arr) {
    if (Array.isArray(item)) {
      result.push(...flatten(item));
    } else {
      result.push(item);
    }
  }

  return result;
}
```
